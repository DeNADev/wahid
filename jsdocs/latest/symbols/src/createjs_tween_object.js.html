 28 /**
 29  * An interface used for controlling a tween.
 30  * @interface
 31  */
 32 createjs.TweenObject = function() {};

 35  * Masks used by the updateTween() method.
 36  * @enum {number}

 40   UPDATE: 4
 41 };

 43 /**
 44  * Updates the position of this tween.

 50 createjs.TweenObject.prototype.updateTween = function(time, flag, next) {};

 52 /**
 53  * Starts playing this tween.
 54  * @param {number} time
 55  */
 56 createjs.TweenObject.prototype.playTween = function(time) {};
 57 
 58 /**
 59  * Stops playing this tween.
 60  * @param {number} time
 61  */
 62 createjs.TweenObject.prototype.stopTween = function(time) {};
 63 
 64 /**
 65  * Sets a proxy that intermediates this tween and its target.
 66  * @param {createjs.TweenTarget} proxy

 68  * @return {number}
 69  */
 70 createjs.TweenObject.prototype.setProxy = function(proxy, targets) {};

 72 /**
 73  * Sets the play offset of this tween.

 77 createjs.TweenObject.prototype.setPosition = function(position, mode) {};
