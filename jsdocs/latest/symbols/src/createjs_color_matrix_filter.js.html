 26 /// <reference path="filter.js"/>

 29 /**
 30  * A class encapsulating a color-matrix filter of CreateJS.
 31  * @param {createjs.ColorMatrix} matrix
 32  * @extends {createjs.Filter}
 33  * @constructor
 34  */
 35 createjs.ColorMatrixFilter = function(matrix) {
 36   createjs.Filter.call(this, createjs.Filter.Type.COLOR_MATRIX);
 37 
 38   /**
 39    * The color matrix used by this filter.
 40    * @type {createjs.ColorMatrix}
 41    * @private
 42    */
 43   this.matrix_ = matrix;
 44 };
 45 createjs.inherits('ColorMatrixFilter',
 46                   createjs.ColorMatrixFilter,
 47                   createjs.Filter);
 48 
 49 /**
 50  * Changes the type of the specified Filter object to createjs.ColorFilter.
 51  * @param {createjs.Filter} filter
 52  * @return {createjs.ColorMatrixFilter}
 53  */
 54 createjs.ColorMatrixFilter.get = function(filter) {
 55   /// <param type="createjs.Filter" name="filter"/>
 56   /// <returns type="createjs.ColorMatrixFilter"/>
 57   return /** @type {createjs.ColorMatrixFilter} */ (filter);
 58 };
 59 
 60 /**
 61  * Returns the color matrix as an array.
 62  * @return {Array.<number>}
 63  */
 64 createjs.ColorMatrixFilter.prototype.getMatrix = function() {
 65   /// <returns type="Array" elementType="number"/>
 66   return this.matrix_.toArray();
 67 };
 68 
 69 // Export the createjs.ColorMatrixFilter class to the global namespace.
 70 createjs.exportObject('createjs.ColorMatrixFilter',
 71                       createjs.ColorMatrixFilter, {
 72 });
