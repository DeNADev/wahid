 29 /**
 30  * A class encapsulating a color filter of CreateJS.
 31  * @param {number=} opt_redMultiplier
 32  * @param {number=} opt_greenMultiplier
 33  * @param {number=} opt_blueMultiplier
 34  * @param {number=} opt_alphaMultiplier
 35  * @param {number=} opt_redOffset
 36  * @param {number=} opt_greenOffset
 37  * @param {number=} opt_blueOffset
 38  * @param {number=} opt_alphaOffset
 39  * @extends {createjs.Filter}
 40  * @constructor
 41  */
 42 createjs.ColorFilter = function(opt_redMultiplier,
 43                                 opt_greenMultiplier,
 44                                 opt_blueMultiplier,
 45                                 opt_alphaMultiplier,
 46                                 opt_redOffset,
 47                                 opt_greenOffset,
 48                                 opt_blueOffset,
 49                                 opt_alphaOffset) {
 50   createjs.Filter.call(this, createjs.Filter.Type.COLOR);
 51 
 52   /**
 53    * The color multiplier.
 54    * @type {createjs.Color}
 55    */
 56   this.multiplier = new createjs.Color(
 57       (opt_redMultiplier != null) ? opt_redMultiplier : 1,
 58       (opt_greenMultiplier != null) ? opt_greenMultiplier : 1,
 59       (opt_blueMultiplier != null) ? opt_blueMultiplier : 1,
 60       (opt_alphaMultiplier != null) ? opt_alphaMultiplier : 1);
 61 
 62   /**
 63    * The color offset.
 64    * @type {createjs.Color}
 65    */
 66   this.offset = new createjs.Color(
 67       opt_redOffset || 0,
 68       opt_greenOffset || 0,
 69       opt_blueOffset || 0,
 70       opt_alphaOffset || 0);
 71 };
 72 createjs.inherits('ColorFilter', createjs.ColorFilter, createjs.Filter);
 73 
 74 /**
 75  * Changes the type of the specified Filter object to createjs.ColorFilter.
 76  * @param {createjs.Filter} filter
 77  * @return {createjs.ColorFilter}
 78  */
 79 createjs.ColorFilter.get = function(filter) {
 80   /// <param type="createjs.Filter" name="filter"/>
 81   /// <returns type="createjs.ColorFilter"/>
 82   return /** @type {createjs.ColorFilter} */ (filter);
 83 };
 84 
 85 // Export the createjs.ColorFilter class to the global namespace.
 86 createjs.exportObject('createjs.ColorFilter', createjs.ColorFilter, {
 87 });
